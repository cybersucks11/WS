Q2) Create a login form with a username and password. Display “Welcome” message if [15M] 
username and password is same otherwise display “Invalid username or password” message. 
<!DOCTYPE html>
<html>
<head>
    <title>Login Form</title>
    <script>
        function checkLogin() {
            var username = document.getElementById("user").value;
            var password = document.getElementById("pass").value;
            if (username === password) {
                document.getElementById("result").innerHTML = "Welcome";
            } else {
                document.getElementById("result").innerHTML = "Invalid username or password";
            }
        }
    </script>
</head>
<body>
    <h2>Login Form</h2>
    Username: <input type="text" id="user"><br><br>
    Password: <input type="password" id="pass"><br><br>
    <button onclick="checkLogin()">Login</button>
    <p id="result"></p>
</body>
</html>
