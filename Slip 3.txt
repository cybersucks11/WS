Q2) Write a simple web service called Dairy Product Price Service. This keeps a Draft I Syllabus Page 44 dairy product price table to record the different kinds of product prices. It will supply 3 services to the client so that clients can add product price, delete product price, update product prize.
Separate files of html , php ,

HTML Form Page (index.html)
<!DOCTYPE html>
<html>
<head>
    <title>Dairy Product Service</title>
</head>
<body>

<h2>Add Product</h2>
<form method="POST" action=" add.php">
    Name: <input type="text" name="product_name" required><br><br>
    Price: <input type="number" step="0.01" name="price" required><br><br>
    <button type="submit">Add Product</button>
</form>

<hr>

<h2>Update Product</h2>
<form method="POST" action="update.php">
    ID: <input type="number" name="id" required><br><br>
    Name: <input type="text" name="product_name" required><br><br>
    Price: <input type="number" step="0.01" name="price" required><br><br>
    <button type="submit">Update Product</button>
</form>

<hr>

<h2>Delete Product</h2>
<form method="POST" action="delete.php">
    ID: <input type="number" name="id" required><br><br>
    <button type="submit">Delete Product</button>
</form>

</body>
</html>

add.php
<?php
include 'db.php';

$product_name = $_POST['product_name'];
$price = $_POST['price'];

$query = "INSERT INTO dairy_products (product_name, price) VALUES ($1, $2)";
$result = pg_query_params($conn, $query, array($product_name, $price));

if ($result) {
    echo "Product added successfully";
} else {
    echo "Error adding product";
}
?>


update.php
<?php
include 'db.php';

$id = $_POST['id'];
$product_name = $_POST['product_name'];
$price = $_POST['price'];

$query = "UPDATE dairy_products SET product_name=$1, price=$2 WHERE id=$3";
$result = pg_query_params($conn, $query, array($product_name, $price, $id));

if ($result) {
    echo "Product updated successfully";
} else {
    echo "Error updating product";
}
?>


delete.php
<?php
include 'db.php';
$id = $_POST['id'];
$query = "DELETE FROM dairy_products WHERE id=$1";
$result = pg_query_params($conn, $query, array($id));
if ($result) {
    echo "Product deleted successfully";
} else {
    echo "Error deleting product";
}
?>
Create Table 
CREATE TABLE dairy_products (
    id SERIAL PRIMARY KEY,
    product_name VARCHAR(100),
    price NUMERIC(10,2)
);
